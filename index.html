// Smooth scrolling for navigation links document.querySelectorAll('nav a').forEach(anchor => { anchor.addEventListener('click', function (e) { e.preventDefault(); const target = document.querySelector(this.getAttribute('href')); target.scrollIntoView({ behavior: 'smooth' }); }); }); // Optional: Add more interactivity, like a dark mode toggle or project hover effects // Example: Simple alert on contact click document.querySelector('#contact a[href^="mailto"]').addEventListener('click', () => { alert('Thanks for reaching out!'); });
